 * scoped
 * 
 * 
 * 带有 scoped 特性的标签仅作用域当前的 component

</script>

 * 
 *
 * * <template><template/> 和 render()，不能共存
 */


render(createElement: CreateElement, context: RenderContext<Props>): VNode | VNode[] {

/**
 * 在 vue 中 template 和 render 的优先级 
 * 
 * 
 */


<template></template>           // 优先级 - 1

export default {

    template: ``,               // 优先级 - 3
    
    render() { },               // 优先级 - 2
}
