#### sitemap.json

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * sitemap.json             // 全局
 * 
 * 
 * https://developers.weixin.qq.com/miniprogram/dev/framework/sitemap.html
 * 
 * 小程序根目录下的 sitemap.json 文件用于配置小程序及其页面是否允许被微信索引
 * 文件内容为一个 JSON 对象，如果没有 sitemap.json ，则默认为所有页面都允许被索引
 */


{
    "rules": [{                     // [{}, {} ...]         // 索引规则列表
    
    
        "action": "allow",          // allow/disallow       // 该规则的页面是否允许被索引     
        
        
        "page": "*",                // * 表示所有页面，不能作为通配符使用
        "page": [                   // []                   // 用于指定部分页面是否允许被所索引
        
            "path/to/page",         // 当 "action": "allow" 时，"path/to/page" 页面允许被索引
            "path/to/page"          // 当 "action": "disallow" 时，"path/to/page" 页面不允许被索引
        ],                 
        
        
        "params": [                 // params 指定索引关键字，根据 matching 所对应的索引规则使用
        
            "a",
            "b",
            "c"
        ],
        
        
        "matching": "exact",        // 当用户输入的索引参数中 "等于" params 时，规则命中
        "matching": "inclusive",    // 当用户输入的索引参数中 "包含" params 时，规则命中
        "matching": "exclusive",    // 当用户输入的索引参数与 params 交集为空时，规则命中
        "matching": "partial",      // 当用户输入的索引参数与 params 交集不为空时，规则命中
        
        
        "priority": "66"            // number               // 优先级，值越大则规则越早被匹配，否则默认从上到下匹配
    }]
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
