#### button

``` javascript

/**
 * button
 * 
 * 
 * @size: string                // 大小
 * @type: string                // 样式
 * 
 * @plain: boolean              // 边框按钮
 * @disable: boolean            // 是否禁用
 * 
 * @hover-class: string         // 指定按钮按下去的样式类。当 hover-class="none" 时，没有点击态效果
 * 
 * ... ...
 */
```

#### input

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * input 
 * 
 * 
 * @value: string                   // 输入框初始内容
 * @type: string                    // 类型 
 * 
 * @password: boolean               // 是否为密码类型
 * 
 * @placeholder: string             // 输入框为空时占位符
 * @placeholder-style: string       // 指定 placeholder 的样式
 * @placeholder-class: string       // 指定 placeholder 的样式类
 * 
 * 
 * @maxlength: number               // 最大输入长度，设置为 -1 时不限制最大长度
 * 
 * ... ...
 */

<input type="" value="" placeholder="" password />


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
