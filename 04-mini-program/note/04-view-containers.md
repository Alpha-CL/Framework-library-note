#### swiper

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * swiper
 * 
 * 
 * @autoplay:boolean                // 是否自动切换
 * 
 * @interval: number                // 自动切换时间间隔
 * @duration: number                // 滑动动画时长
 * 
 * @indicator-dots: boolean         // 是否显示面板指示点
 * @indicator-color: boolean        // 指示点颜色
 * 
 * @current: number                 // 根据 num( swiper-item 的索引 )，显示指定索引的子项
 * 
 * @vertical: boolean               // 轮播滑动方向，若要纵向，则需要设置 swiper 高度
 */

<swiper autoplay interval="5000" duration="300" indicator-dots >
    <swiper-item url=""></swiper-item>
    <swiper-item></swiper-item>
    <swiper-item></swiper-item>
</swiper>

//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//



<view>
    <!-- 轮播垂直： style="height: 200rpx" vertical -->

    <!-- interval: 自动切换间隔 -->
    <!-- duration: 滑动时常 -->
    <!-- Indicator-dot: 显示图片指示点 -->
    <!-- bindchange="" 当轮播图滑动时触发 -->
    <swiper autoplay interval="5000" duration="300" indicator-dots>
        <!-- url: 指定跳转地址，若指定页面需从根目录开始( '/' )-->
```
